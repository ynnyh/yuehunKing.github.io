```
git config --config user.name 'yuehun' // 设置全局用户名
git config --config user.email '1234@qq.com' // 设置邮箱
git mv readme readme.md // git重命名方法
git reset --hard // git移除所有暂存区内容
git log --all // 查看所有分支下的commit
git log --all --graph // 图形化查看所有分支下的commit
git log --online // 一行简洁的方式查看当前分支下的commit
git log --online --all // 一行简介的方式查看所有分支下的commit
git log --online --all -n4 // 查看就近的四个commit，n后面跟数字多少就查多少
git log --online --all -n4 --graph // 图形化查看就近的四个commit
gitk // 图形界面工具查看版本历史
```
